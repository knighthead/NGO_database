             function loadOrg( id )
{
	if( !document.getElementById  )
	{
		cpopUp('org_details.aspx?orgID=' + id );
	}
	else 
	{
		var f = document.getElementById( "loader" );
		if( f.innerHTML == "" || !f.contentWindow ) cpopUp('org_details.aspx?orgID=' + id );
		else
		{
			//if already loaded, hide
			var t = document.getElementById( "d" + id );
			if( t.style.display == "block" ) hideOrgDetails( id );
			else 
			{
				document.getElementById( "s" + id ).style.display="none";
				document.getElementById( "l" + id ).style.display="block";
				f.src = "org_details.aspx?inline=1&orgID=" + id; //load data
			}
		}
	}
}

function onLoadOrg( id )
{
	document.getElementById( "l" + id ).style.display="none";
	document.getElementById( "s" + id ).style.display="none";
	var t = document.getElementById( "d" + id );
	var f = document.getElementById( "loader" );
	t.innerHTML = f.contentWindow.document.getElementById( "content" ).innerHTML;
	t.style.display = "block";
}
function hideOrgDetails( id )
{
	document.getElementById( "d" + id ).style.display="none";
	document.getElementById( "s" + id ).style.display="block";
}
